PROGRAM MENENTUKAN_BILANGAN_PRIMA

DEKLARASI 
variable BILANGAN, I : integer

DESKRIPSI 

INPUT BILANGAN
IF (BILANGAN <= 1) THEN
	OUTPUT BUKAN BILANGAN PRIMA

ELSE
	I = 2

	WHILE I < BILANGAN THEN
		IF (BILANGAN % I == 0) THEN 
			OUTPUT BUKAN BILANGAN PRIMA 
		ELSE
			I = I + 1
	ELSE 
		OUTPUT BILANGAN PRIMA
