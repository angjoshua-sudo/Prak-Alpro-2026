MENGHITUNG_FAKTORIAL

DEKLARASI
variable nilai_faktorial, faktorial, I : integer

DESKRIPSI
INPUT nilai_faktorial

IF nilai_faktorial< 0 THEN
	OUTPUT "TIDAK BOLEH NEGATIF"
ELSE
	faktorial = 1
	I = 1
	
	WHILE I <= nilai_faktorial DO
		faktorial = faktorial*I
		I = I + 1
	ENDWHILE
	OUTPUT faktorial
ENDIF
