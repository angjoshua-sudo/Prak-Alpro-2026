MENENTUKAN AKAR_PERSAMAAN_KUADRAT


DEKLARASI 
variable koefisien_x^2, koefisien_x, konstanta : INTEGER

DESKRIPSI
INPUT koefisien_x^2 
INPUT koefisein_x
INPUT konstanta

D = (koefisien_x)**2 - 4*(koefisen_x^2)*(konstanta)

IF D > 0 THEN
	OUTPUT "AKAR REAL DAN BERLAWANAN"
ELSE
	IF D = 0 THEN
		OUTPUT "AKAR REAL DAN KEMBAR"
	ELSE
		IF D < 0 
			OUTPUT "AKAR IMAJINER / TIDAK REAL / KHAYAL"
		ELSE
			OUTPUT "AKAR REAL / NYATA"
		ENDIF
	ENDIF
ENDIF
